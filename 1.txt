const readline = require('readline').createInterface({
    input: process.stdin,
    output: process.stdout
});

// Функция для сравнения двух строк
function compareStrings(a, b) {
    if (a === b) {
        console.log("Строки идентичны");
    } else {
        console.log("Строки неидентичны");
    }
}

// Читаем первую строку
readline.question('Введите первую строку (a): ', (a) => {
    // Читаем вторую строку
    readline.question('Введите вторую строку (b): ', (b) => {
        // Сравниваем строки и выводим результат
        compareStrings(a, b);

        // Закрываем интерфейс ввода
        readline.close();
    });
});
